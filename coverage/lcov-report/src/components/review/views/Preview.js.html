          primaryText: this.state.verified[0],
          secondaryText: this.state.verified[1],

        break;

        this.setState({
          primaryText: this.state.rejected[0],
          secondaryText: this.state.rejected[1],
        });
        break;

          primaryText: "Move to verified",
          secondaryText: "Reject",
        });
